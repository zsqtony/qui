export const DEUXSTag = ({ label, bgColor = "#1677ff" }) => (
  <span
    style={{
      padding: "3px 6px",

#### 创建 React tsx 文件后：直接复制代码可用

```jsx
export default function Tag({ label, bgColor = "#1677ff" }) {
  return (
    <span
      style={{
        padding: "3px 6px",
        fontSize: "12px",
        backgroundColor: bgColor,
        color: "#fff",
        borderRadius: "3px",
      }}
    >
      {label}
    </span>
  );
}
```

```jsx

  return (

        fontSize: "12px",

        color: "#fff",

      {label}
    </span>

```

#### 背景变为透明背景样式

export const TYPEXSTag = ({
  label,
  type = "dark",
  shape = "3px",
  bgColor = "#1677ff",
}) => (
  <span
    style={{
      padding: "3px 6px",
      fontSize: "12px",
      backgroundColor: type == "dark" ? bgColor : "",
      color: type == "dark" ? "#fff" : bgColor,
      borderRadius: shape,
      border: `1px solid ${bgColor}`,
    }}
  >
    {label}
  </span>
);

<TYPEXSTag label="在岸的对面" type="plain" />
<br />
<TYPEXSTag label="在岸的对面" type="plain" shape="100px" bgColor="#FF9416" />

#### 增加一种类型：type="plain"

```jsx
export default function Tag({
  label,
  type = "dark",
  shape = "3px",
  bgColor = "#1677ff",
}) {
  return (
    <span
      style={{
        padding: "3px 6px",
        fontSize: "12px",
        backgroundColor: type == "dark" ? bgColor : "",
        color: type == "dark" ? "#fff" : bgColor,
        borderRadius: shape,
        border: `1px solid ${bgColor}`,
      }}
    >
      {label}
    </span>
  );
}
```

最后简单的写了下判断，这样一个简单且比较完善的 Tag 标签就写好了，可以直接复制粘贴到自己项目里，直接成为你的组件<br />
当然了还有优化的空间，你可以继续优化改动他

#### API

#### Props

| 属性  | 说明         | 类型   | 默认值 | 可选值                                                                              |
| ----- | ------------ | ------ | ------ | ----------------------------------------------------------------------------------- |
| label | 文字内容     | String | -      | -                                                                                   |
| shape | 设置圆角     | String | 3px    | 根据实际开发自行设置值，如果是用于开发小程序可使用：2 倍的 rpx 为单位<br />如：6rpx |
| plain | 镂空透明背景 | String | dark   | dark/plain                                                                          |
