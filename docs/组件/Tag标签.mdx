# Tag 标签

export const DEUXSTag = ({ label, bgColor = "#1677ff" }) => (
  <span
    style={{
      padding: "3px 6px",
      fontSize: "12px",
      backgroundColor: bgColor,
      color: "#fff",
      borderRadius: "3px",
    }}
  >
    {label}
  </span>
);

#### 简单示例：

<DEUXSTag label="附近地铁" />
<br />
<DEUXSTag label="附近商场" bgColor="#FF9416" />
<br />
<DEUXSTag label="附近学校" bgColor="#000" />

#### 创建 React tsx 文件后：直接复制代码可用

```jsx
export default function Tag({ label, bgColor = "#1677ff" }) {
  return (
    <span
      style={{
        padding: "3px 6px",
        fontSize: "12px",
        backgroundColor: bgColor,
        color: "#fff",
        borderRadius: "3px",
      }}
    >
      {label}
    </span>
  );
}
```

#### 背景变为透明模式

export const TYPEXSTag = ({
  label,
  type = "dark",
  shape = "3px",
  bgColor = "#1677ff",
}) => (
  <span
    style={{
      padding: "3px 6px",
      fontSize: "12px",
      backgroundColor: type == "dark" ? bgColor : "",
      color: type == "dark" ? "#fff" : bgColor,
      borderRadius: shape,
      border: `1px solid ${bgColor}`,
    }}
  >
    {label}
  </span>
);

<TYPEXSTag label="在岸的对面" type="plain" />
<br />
<TYPEXSTag label="在岸的对面" type="plain" shape="100px" bgColor="#FF9416" />

#### 增加一种类型：type="plain"

```jsx
export default function Tag({
  label,
  type = "dark",
  shape = "3px",
  bgColor = "#1677ff",
}) {
  return (
    <span
      style={{
        padding: "3px 6px",
        fontSize: "12px",
        backgroundColor: type == "dark" ? bgColor : "",
        color: type == "dark" ? "#fff" : bgColor,
        borderRadius: shape,
        border: `1px solid ${bgColor}`,
      }}
    >
      {label}
    </span>
  );
}
```
